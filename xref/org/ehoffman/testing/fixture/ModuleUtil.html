14    /**
15     * @param provider a class should implement ModuleGroup or Module, not just ModuleProvider, which acts a common marking super interface.
16     * A module represents a single, atomic piece of a test fixture.
17     * A module group contains a list of class to ModuleProviders, which may in turn be module groups.
18     * 
19     * This method walks that chain an returns the list of all Modules referenced by a ModuleGroup
20     * 
21     * @return a List of a Module Class, that are referenced by the ModuleGroup, or the input if it is an implementation of Module and not ModuleProvider
22     * @throws IllegalAccessException
23     * @throws InstantiationException
24     */
25    public static Set<Class<? extends Module<?>>> getAllPossibleModules(Class<? extends ModuleProvider<?>> provider) {
26      Set<Class<? extends Module<?>>> modules = new HashSet<Class<? extends Module<?>>>();
27      if (ModuleGroup.class.isAssignableFrom(provider)){
28        try {
29          List<Class<? extends ModuleProvider<?>>> classes = ((ModuleGroup<?>)provider.newInstance()).getModuleClasses();
30          if (classes != null){
31            for (Class<? extends ModuleProvider<?>> _provider : classes){
32              modules.addAll(getAllPossibleModules(_provider));
33            }
34          }
35        } catch (IllegalAccessException e){
36          throw new RuntimeException("a class that implements ModuleGroup must have a public, no argument constructor"+provider, e);
37        } catch (InstantiationException e){
38          throw new RuntimeException("a class that implements ModuleGroup could not be instanciated "+provider, e);
39        }
40      } else if (Module.class.isAssignableFrom(provider)){
41        @SuppressWarnings("unchecked")
42        Class<? extends Module<?>> clazz = (Class<? extends Module<?>>) provider;
43        modules.add(clazz);
44      } else {
45        throw new RuntimeException("only an instance of module or modulegroup may be used to configure a foctory "+provider);
46      }
47      return modules;
48    }

50    public static Set<Class<? extends Module<?>>> getAllPossibleModules(Collection<Class<? extends ModuleProvider<?>>> providers) {
51      Set<Class<? extends Module<?>>> modules = new HashSet<Class<? extends Module<?>>>();
52      for (Class<? extends ModuleProvider<?>> provider : providers){
53        modules.addAll(getAllPossibleModules(provider));

55      return modules;
