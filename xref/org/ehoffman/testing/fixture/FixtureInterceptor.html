33    private Iterator<Set<Class<? extends Module<?>>>> fixtureIterator(Fixture fixture) {
34      Class<? extends ModuleProvider<?>>[] moduleArray = fixture.factory();
35      return FixtureContainer.getDotProductModuleCombinations(Arrays.asList(moduleArray), fixture.destructive());
36    }

38    /**
39     * 
40     * @param instance the inputed IMethodInstance that has not yet been evaluated
41     * @param context
42     * @return a set containing either the IMethodInstance that was passed in, or set or ITestMethods with a corresponding key in the {@link FixtureInterceptor#testNGMethodToSetOfModuleClassesForSingleInvocation}
43     * map that has a value of the Set or modules that particular instance of IMethodInstance should be run with.  The services the {@link Module}s provide with be available via the {@link FixtureContainer} to the test.
44     */
45    private List<IMethodInstance> calculateRuns(IMethodInstance instance){
46      List<IMethodInstance> output = new ArrayList<IMethodInstance>();
47      Method method = instance.getMethod().getConstructorOrMethod().getMethod();
48      if (method != null && method.getAnnotation(Fixture.class) != null) {
49        Fixture fixture = method.getAnnotation(Fixture.class);
50        Iterator<Set<Class<? extends Module<?>>>> fixtureIterator = fixtureIterator(fixture);
51        while (fixtureIterator.hasNext()) {
52          ITestNGMethod testNGMethod = instance.getMethod().clone();
53          IMethodInstance newMultiInstance = new MethodInstance(testNGMethod);
54          testNGMethodToSetOfModuleClassesForSingleInvocation.put(testNGMethod, fixtureIterator.next());
55          logger.info("added "+testNGMethod.toString()+" with modules of "+testNGMethodToSetOfModuleClassesForSingleInvocation.get(testNGMethod));
56          output.add(newMultiInstance);
57        }
58      } else {
59        output.add(instance);
60      }
61      logger.info("added all "+testNGMethodToSetOfModuleClassesForSingleInvocation);
62      return output;
63    }

67    @Override
68    public List<IMethodInstance> intercept(List<IMethodInstance> methods) {
69      List<IMethodInstance> output = new ArrayList<IMethodInstance>();
70      for (IMethodInstance instance : methods){
71        output.addAll(calculateRuns(instance));
72      }
73      return output;
74    }

86      ITestNGMethod testNGmethod = testResult.getMethod();
87      Set<Class<? extends Module<?>>> moduleClasses = testNGMethodToSetOfModuleClassesForSingleInvocation.get(testNGmethod);
88      if (moduleClasses != null){
89        FixtureContainer.setModuleClasses(moduleClasses, testNGmethod.getConstructorOrMethod().getMethod().getAnnotation(Fixture.class).destructive());
90      } else {
91        FixtureContainer.setModuleClasses(emptySet, false);
92        FixtureContainer.wipeFixture();
93      }

100       Collections.sort(names);

102     }
