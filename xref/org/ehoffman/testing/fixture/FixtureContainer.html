53    private static Module<?> getModuleFromClass(Class<? extends Module<?>> moduleClass, boolean destructive) {
54      Module<?> module = null;
55      if (!destructive) {
56        module = reusableThreadSafeModules.get(moduleClass);
57        if (module == null) {
58          module = getNewInstance(moduleClass);
59          reusableThreadSafeModules.putIfAbsent(moduleClass, module);
60        }
61        if (module != reusableThreadSafeModules.get(moduleClass)) {
62          module.destroy();
63          module = reusableThreadSafeModules.get(moduleClass);
64        }
65      } else {
66        module = getNewInstance(moduleClass);
67      }
68      return module;
69    }

77    static Collection<Set<Class<? extends Module<?>>>> mergeListsOfSameModuleType(Collection<Class<? extends Module<?>>> input) {
78      Map<String, Set<Class<? extends Module<?>>>> output = new HashMap<String, Set<Class<? extends Module<?>>>>();
79      if (input != null) {
80        for (Class<? extends Module<?>> clazz : input) {
81          Module<?> module = getModuleFromClass(clazz, false);
82          String type = module.getModuleType();
83          if (output.get(type) == null) {
84            Set<Class<? extends Module<?>>> newList = new HashSet<Class<? extends Module<?>>>();
85            newList.add(clazz);
86            output.put(type, newList);
87          } else {
88            output.get(type).add(clazz);
89          }
90        }
91      }
92      return output.values();
93    }
