41    public BrokenInterceptor(Boolean runKnownBreaks, Class<? extends Annotation> brokenAnnotation, boolean ideMode) {
42      run_known_breaks = runKnownBreaks;
43      knownBreakClass = brokenAnnotation;
44      ideMode = Boolean.valueOf(System.getProperty("java.class.path").contains("org.testng.eclipse"));
45      logger.info("");

48    private void postProcessBrokenTests(ITestResult testResult) {
49      Method method = testResult.getMethod().getConstructorOrMethod().getMethod();
50      if (method != null){
51        Broken broken = method.getAnnotation(Broken.class);
52        if (broken != null) {
53          testResult.setAttribute("Known Break", true);
54          testResult.setAttribute("True Status", "SUCCESS");
55          if (!testResult.isSuccess() && testResult.getStatus()!=ITestResult.SKIP) {
56            testResult.setAttribute("True Status", "FAILURE");
57            testResult.setStatus(ITestResult.SKIP);
58            testResult.setThrowable(null);
59          }
60        }
61      }
62    }
63  
64    private boolean shouldExecuteMethod(Method method) {
65      verifyBrokenAnnotation(method.getAnnotation(knownBreakClass));;
66      return (ideMode || run_known_breaks || method.getAnnotation(knownBreakClass) == null);
67    }
