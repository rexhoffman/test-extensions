59  public class ExtensibleTestNGListener implements IMethodInterceptor, IInvokedMethodListener, ITestListener {
60    
61    private static Logger logger = LoggerFactory.getLogger(ExtensibleTestNGListener.class);
62    
63    private static Map<Class<? extends ExtensibleTestNGListener>, List<? extends Interceptor>> clazzToInterceptors = new HashMap<Class<? extends ExtensibleTestNGListener>, List<? extends Interceptor>>();
64    
65    public static void setInterceptors(Class<? extends ExtensibleTestNGListener> clazz, List<? extends Interceptor> interceptors){
66      clazzToInterceptors.put(clazz, interceptors);
67    }
68    
69    private List<? extends Interceptor> getInterceptors(){
70      return clazzToInterceptors.get(this.getClass());
71    }
72    
73    @Override
74    public List<IMethodInstance> intercept(List<IMethodInstance> methods, ITestContext context){
75      List<IMethodInstance> output = methods;
76      for (Interceptor interceptor : getInterceptors()){
77        output = interceptor.intercept(output);
78      }
79      return output;
80    }

83    @Override
84    public void beforeInvocation(IInvokedMethod method, ITestResult testResult) {

92    public void afterInvocation(IInvokedMethod method, ITestResult testResult) {

100   @Override
101   public void onFinish(ITestContext context) {
102     StringBuilder errors = new StringBuilder();
103     boolean error = false;
104     for (Interceptor interceptor : getInterceptors()){
105       for (String message : interceptor.getConfigErrorMessages()){
106         errors.append(message).append("\n");
107         error = true;
108       }
109     }
110     for (Interceptor interceptor : getInterceptors()){
111       interceptor.shutdown();
112     }
113     if (error){
114       logger.error(errors.toString());
115       throw new RuntimeException("Tests are not properly configured "+errors);
116     }
