26     * Start capturing.
27     * defaults to the root logger
28     * defaults to INFO level
29     * defaults to "[%p] %m%n" layout pattern
30     */
31    public static void start() {
32      if (INSTANCE.get() != null) {
33        throw new IllegalStateException("already started");
34      }
35      INSTANCE.set(new LogbackCapture("", Level.INFO, "[%p] %c.%M %m\n"));
36    }

38    /**
39     * Start capturing.
40     * @param loggerName if null, defaults to the root logger
41     * @param level if null, defaults to all levels
42     * @param layoutPattern if null, defaults to "[%p] %m%n"
43     */
44    public static void start(String loggerName, Level level, String layoutPattern) {
45      if (INSTANCE.get() != null) {
46        throw new IllegalStateException("already started");
47      }
48      INSTANCE.set(new LogbackCapture(loggerName, level, layoutPattern));
49    }

51    /** Stop capturing and return the logs. */
52    public static String stop() {
53      LogbackCapture instance = INSTANCE.get();
54      if (instance == null) {
55        throw new IllegalStateException("was not running");
56      }
57      final String result = instance.stopInstance();
58      INSTANCE.remove();
59      return result;
60    }

69    private LogbackCapture(String loggerName, Level level, String layoutPattern) {
70      logs = new ByteArrayOutputStream(4096);
71      encoder = buildEncoder(layoutPattern);
72      appender = buildAppender(encoder, logs);
73      logger = getLogbackLogger(loggerName, level);
74      logger.addAppender(appender);
75    }

88    private static Logger getLogbackLogger(String name, Level level) {
89      if (name == null || "".equals(name)) {
90        name = ROOT_LOGGER_NAME;
91      }
92      if (level == null) {
93        level = ALL;
94      }
95  
96      //resets logger context
97      LoggerContext loggerContext =
98          (LoggerContext) LoggerFactory.getILoggerFactory();
99      if (!loggerContext.isStarted()){
100       loggerContext.start();
101     }
102     //loggerContext.stop();
103 
104     loggerContext.getLogger(name);
105 
106     Logger logger = ContextSelectorStaticBinder.getSingleton()
107         .getContextSelector().getDefaultLoggerContext().getLogger(name);
108     logger.setLevel(level);
109     return logger;
110   }

112   private static Encoder<ILoggingEvent> buildEncoder(String layoutPattern) {
113     if (layoutPattern == null) {
114       layoutPattern = "[%p] %c.%M %m\n";
115     }
116     PatternLayoutEncoder encoder = new PatternLayoutEncoder();
117     encoder.setPattern(layoutPattern);
118     //encoder.setCharset(Charset.forName("UTF-16"));
119     if (ContextSelectorStaticBinder.getSingleton() != null &&
120         ContextSelectorStaticBinder.getSingleton().getContextSelector() != null){
121       encoder.setContext(ContextSelectorStaticBinder.getSingleton().getContextSelector().getDefaultLoggerContext());
122     }
123     encoder.start();
124     return encoder;
125   }

127   private static OutputStreamAppender<ILoggingEvent> buildAppender(final Encoder<ILoggingEvent> encoder,
128       final OutputStream outputStream) {
129     OutputStreamAppender<ILoggingEvent> appender = new OutputStreamAppender<ILoggingEvent>();
130     appender.setName("logcapture");
131     if (ContextSelectorStaticBinder.getSingleton() != null &&
132         ContextSelectorStaticBinder.getSingleton().getContextSelector() != null){
133       appender.setContext(ContextSelectorStaticBinder.getSingleton().getContextSelector().getDefaultLoggerContext());
134     }
135     appender.setEncoder(encoder);
136     appender.setOutputStream(outputStream);
137     appender.addFilter( new Filter<ILoggingEvent>() {
138       long thread = Thread.currentThread().getId();
139       @Override
140       public FilterReply decide(ILoggingEvent event) {
141         return (thread==Thread.currentThread().getId())?FilterReply.ACCEPT:FilterReply.DENY;
142       }
143     });
144     appender.start();
145     return appender;
146   }
