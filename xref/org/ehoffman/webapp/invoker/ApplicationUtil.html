34    /**
35     * Used to detect applications that may be run or accessed from a test, either in IDE, or via command line.
36     * 
37     * @param classes the list order determines order of precedence, the first @{ApplicationLookUpMethod} to find an Application with a matching name, will be used if that application is used.
38     * @return A set of Applications, ready to be passed to {@link #runApplicationOnOwnServer(Application)}
39     */
40    public static Set<Application> discoverApplications(List<Class<? extends ApplicationLookUpMethod>> classes) {
41      Set<Application> applications = new HashSet<Application>();
42      for (Class<? extends ApplicationLookUpMethod> clazz : classes){
43        ApplicationLookUpMethod lookUpMethod = null;
44        try {
45          lookUpMethod = clazz.newInstance();
46        } catch (IllegalAccessException e){
47          throw new RuntimeException(e);
48        } catch (InstantiationException e){
49          throw new RuntimeException(e);
50        }
51        applications.addAll(lookUpMethod.lookup()); //set addAll skips duplicates (which in the case of applications is based on the context root)
52      }
53      return applications;
54    }

77        ApplicationLookUpMethod lookUpMethod = null;
