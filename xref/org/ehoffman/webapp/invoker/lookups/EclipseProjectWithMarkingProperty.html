26    private File[] findDefaultContentLocations(File projectDir){
27      List<File> output = new ArrayList<File>();
28      for (String location : defaultLocations){
29        File file = new File(projectDir, location);
30        if (file != null && file.exists() && file.isDirectory()){
31          output.add(file);
32        }
33      }
34      return output.toArray(new File[output.size()]);
35    }

38    @Override
39    public List<Application> lookup() {
40      Enumeration<URL> urls = null;
41      try {
42        urls = ApplicationUtil.class.getClassLoader().getResources("webapp.properties");
43      } catch (IOException io_exception){
44        throw new RuntimeException("could not determine file of resource, while try to calculate project location", io_exception);
45      }
46      List<Application> applications = new ArrayList<Application>();
47      for (URL url : Collections.list(urls)){
48        try {
49          logger.info("started by "+url);
50          File basetxt = new File(url.toURI());
51          Properties props = new Properties();
52          props.load(new FileReader(basetxt));
53          File projectBase =  basetxt.getParentFile().getParentFile().getParentFile();
54          logger.info("running for "+projectBase);
55          logger.info("Context root "+props.getProperty("contextRoot"));
56          applications.add(Application.buildExploded(findDefaultContentLocations(projectBase)).setContextRoot(props.getProperty("contextRoot")).build());
57        } catch (URISyntaxException exception){
58          throw new RuntimeException("could not determine file of resource, while try to calculate project location");
59        } catch (IOException io_exception){
60          throw new RuntimeException("could not determine file of resource, while try to calculate project location", io_exception);
61        }
62      }
63      return applications;
64    }

66    @Override
67    public Application lookupByName(String contextRoot) {
68      for (Application application : lookup()){
69        if (contextRoot != null && contextRoot.equals(application.getContextRoot())){
70          return application;
71        }
72      }
73      return null;
74    }
