19    @Override
20    public List<Application> lookup() {
21      File file = calculateTargetDirectory();
22      String[] warsNames = file.list();
23      List<Application> applications = new ArrayList<Application>();
24      for (String fileName : warsNames) {
25        if (fileName.toLowerCase().endsWith(".war")) {
26          applications.add(Application.buildWar(new File(file, fileName)).build());
27        }
28      }
29      return applications;
30    }

32    @Override
33    public Application lookupByName(String contextRoot) {
34      File file = calculateTargetDirectory();
35      logger.info("Warfile root is: " + file.getAbsolutePath());
36      String[] warsNames = file.list();
37      for (String name : warsNames) {
38        logger.info("looking at: " + name);
39        if (name.toLowerCase().contains(contextRoot.toLowerCase()) && name.toLowerCase().endsWith(".war")) {
40          return Application.buildWar(new File(file, name)).build();
41        }
42      }
43      return null;
44    }
